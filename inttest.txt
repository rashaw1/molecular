#include <iostream>
#include <fstream>
#include "integrals.hpp"
#include "logger.hpp"
#include "molecule.hpp"
#include "pbf.hpp"
#include "vector.hpp"
#include "matrix.hpp"

int main(int argc, char* argv[])
{
  std::ifstream input("inputfile.mol");
  std::ofstream output("inty.out");
  Logger log(input, output, std::cout);

  PBF p1(1.0, 2, 1, 1);
  PBF p2(1.5, 1, 1, 2);
  PBF p3(1.3, 2, 1, 1);
  PBF p4(1.9, 1, 1, 2);

  Vector c1(3, 0.0); Vector c2(3, 0.0);
  c2[2] = 1.0;
  Molecule mol(log);
  IntegralEngine ints(mol);

  Matrix pMat;

  pMat = ints.twoe(p1, p2, p3, p4, c1, c2, c2, c1);
  pMat.print();
  
  return 0;
}
